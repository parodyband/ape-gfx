package gfx
	constants
		Binding_Group_Invalid :: Binding_Group(0)
		Binding_Group_Layout_Invalid :: Binding_Group_Layout(0)
		Buffer_Invalid :: Buffer(0)
			Invalid handle sentinels returned when resource creation fails.

		COLOR_MASK_A :: u8(1 << 3)
		COLOR_MASK_B :: u8(1 << 2)
		COLOR_MASK_G :: u8(1 << 1)
		COLOR_MASK_R :: u8(1 << 0)
		COLOR_MASK_RGB :: COLOR_MASK_R | COLOR_MASK_G | COLOR_MASK_B
		COLOR_MASK_RGBA :: COLOR_MASK_RGB | COLOR_MASK_A
		Compute_Pipeline_Invalid :: Compute_Pipeline(0)
		Image_Invalid :: Image(0)
		MAX_BINDING_GROUPS :: 8
		MAX_BINDING_GROUP_ENTRIES :: MAX_SHADER_BINDINGS
		MAX_COLOR_ATTACHMENTS :: 8
		MAX_IMAGE_MIPS :: 16
		MAX_RESOURCE_VIEWS :: 32
		MAX_SAMPLERS :: 16
		MAX_SHADER_BINDINGS :: 64
		MAX_UNIFORM_BLOCKS :: 16
		MAX_VERTEX_ATTRIBUTES :: 16
		MAX_VERTEX_BUFFERS :: 8
		Pipeline_Invalid :: Pipeline(0)
		Pipeline_Layout_Invalid :: Pipeline_Layout(0)
		Sampler_Invalid :: Sampler(0)
		Shader_Invalid :: Shader(0)
		View_Invalid :: View(0)

	procedures
		apply_binding_group :: proc(ctx: ^Context, group: Binding_Group, base_bindings: Bindings = {}) -> bool {...}
			apply_binding_group validates one object-backed binding group against the active pipeline and applies it with optional geometry bindings.

		apply_binding_groups :: proc(ctx: ^Context, groups: []Binding_Group, base_bindings: Bindings = {}) -> bool {...}
			apply_binding_groups validates all object-backed binding groups against the active pipeline and applies them with optional geometry bindings.

		apply_pipeline :: proc(ctx: ^Context, pipeline: Pipeline) -> bool {...}
			apply_pipeline binds a graphics pipeline inside a render pass.

		apply_uniform :: proc(ctx: ^Context, group: u32, slot: int, value: ^$T) -> bool {...}
			apply_uniform uploads a typed uniform value using size_of(T).

		apply_uniforms :: proc(ctx: ^Context, group: u32, slot: int, data: Range) -> bool {...}
			apply_uniforms uploads one reflected uniform block to the current pipeline.

		backend_name :: proc(backend: Backend) -> string {...}
			backend_name returns a stable lowercase display name for a Backend.

		begin_compute_pass :: proc(ctx: ^Context, desc: Compute_Pass_Desc = {}) -> bool {...}
			begin_compute_pass starts a compute-only pass.

		begin_pass :: proc(ctx: ^Context, desc: Pass_Desc) -> bool {...}
			begin_pass starts a graphics render pass.

		binding_group_layout_valid :: proc(layout: Binding_Group_Layout) -> bool {...}

		buffer_valid :: proc(buffer: Buffer) -> bool {...}

		compute_pipeline_valid :: proc(pipeline: Compute_Pipeline) -> bool {...}
			compute_pipeline_valid reports whether a Compute_Pipeline handle is nonzero.

		create_binding_group :: proc(ctx: ^Context, desc: Binding_Group_Desc) -> (: Binding_Group, : bool) {...}
			create_binding_group creates an immutable binding group from a generated layout handle and resource handles.

		create_binding_group_layout :: proc(ctx: ^Context, desc: Binding_Group_Layout_Desc) -> (: Binding_Group_Layout, : bool) {...}
			create_binding_group_layout creates an immutable generated binding group layout handle.

		create_buffer :: proc(ctx: ^Context, desc: Buffer_Desc) -> (: Buffer, : bool) {...}
			create_buffer creates a GPU buffer and reports whether creation succeeded.
			On failure, the returned handle is Buffer_Invalid and last_error explains why.

		create_compute_pipeline :: proc(ctx: ^Context, desc: Compute_Pipeline_Desc) -> (: Compute_Pipeline, : bool) {...}
			create_compute_pipeline creates an immutable compute pipeline state object.
			On failure, the returned handle is Compute_Pipeline_Invalid and last_error explains why.

		create_image :: proc(ctx: ^Context, desc: Image_Desc) -> (: Image, : bool) {...}
			create_image creates a texture, storage image, or attachment image.
			On failure, the returned handle is Image_Invalid and last_error explains why.

		create_pipeline :: proc(ctx: ^Context, desc: Pipeline_Desc) -> (: Pipeline, : bool) {...}
			create_pipeline creates an immutable graphics pipeline state object.
			On failure, the returned handle is Pipeline_Invalid and last_error explains why.

		create_pipeline_layout :: proc(ctx: ^Context, desc: Pipeline_Layout_Desc) -> (: Pipeline_Layout, : bool) {...}
			create_pipeline_layout creates an immutable layout composed from generated binding group layouts.

		create_sampler :: proc(ctx: ^Context, desc: Sampler_Desc) -> (: Sampler, : bool) {...}
			create_sampler creates immutable texture sampling state.
			On failure, the returned handle is Sampler_Invalid and last_error explains why.

		create_shader :: proc(ctx: ^Context, desc: Shader_Desc) -> (: Shader, : bool) {...}
			create_shader creates backend shader objects from compiled shader bytecode.
			On failure, the returned handle is Shader_Invalid and last_error explains why.

		create_view :: proc(ctx: ^Context, desc: View_Desc) -> (: View, : bool) {...}
			create_view creates one sampled, storage, color, or depth-stencil view.
			On failure, the returned handle is View_Invalid and last_error explains why.

		default_pass_action :: proc() -> Pass_Action {...}
			default_pass_action returns clear/store defaults for color, depth, and stencil attachments.

	types
		Backend :: enum {Auto, Null, D3D11, Vulkan}
			Backend selects the native graphics implementation used by a Context.

		Binding_Group :: distinct u64
		Binding_Group_Desc :: struct {label: string, layout: Binding_Group_Layout, views: [MAX_RESOURCE_VIEWS]View, samplers: [MAX_SAMPLERS]Sampler}
			Binding_Group_Desc creates an object-backed binding group from a generated layout handle.

		Binding_Group_Layout :: distinct u64
		Binding_Group_Layout_Desc :: struct {label: string, group: u32, entries: [MAX_BINDING_GROUP_ENTRIES]Binding_Group_Layout_Entry_Desc, native_bindings: [MAX_SHADER_BINDINGS]Binding_Group_Native_Binding_Desc}
			Binding_Group_Layout_Desc is generated from Slang reflection and creates Binding_Group_Layout handles.

		Binding_Group_Layout_Entry_Desc :: struct {active: bool, stages: Shader_Stage_Set, kind: Shader_Binding_Kind, slot: u32, name: string, uniform_block: Binding_Group_Uniform_Block_Layout_Desc, resource_view: Binding_Group_Resource_View_Layout_Desc}
			Binding_Group_Layout_Entry_Desc describes one logical entry in a generated binding group.

		Binding_Group_Native_Binding_Desc :: struct {active: bool, target: Backend, stage: Shader_Stage, kind: Shader_Binding_Kind, slot: u32, native_slot: u32, native_space: u32}
			Binding_Group_Native_Binding_Desc maps one logical binding entry to a backend native slot.

		Binding_Group_Resource_View_Layout_Desc :: struct {view_kind: View_Kind, access: Shader_Resource_Access, storage_image_format: Pixel_Format, storage_buffer_stride: u32}
			Binding_Group_Resource_View_Layout_Desc describes one reflected resource view entry.

		Binding_Group_Uniform_Block_Layout_Desc :: struct {size: u32}
			Binding_Group_Uniform_Block_Layout_Desc describes one reflected uniform block entry.

		Bindings :: struct {vertex_buffers: [MAX_VERTEX_BUFFERS]Buffer_Binding, index_buffer: Buffer_Binding, views: [MAX_BINDING_GROUPS][MAX_RESOURCE_VIEWS]View, samplers: [MAX_BINDING_GROUPS][MAX_SAMPLERS]Sampler}
			Bindings contains transient resources applied before draw or dispatch.

		Blend_Factor :: enum {Default, Zero, One, Src_Color, One_Minus_Src_Color, Src_Alpha, One_Minus_Src_Alpha, Dst_Color, One_Minus_Dst_Color, Dst_Alpha, One_Minus_Dst_Alpha, Blend_Color, One_Minus_Blend_Color, Src_Alpha_Saturated}
			Blend_Factor selects source/destination blend factors. Default uses backend default state.

		Blend_Op :: enum {Default, Add, Subtract, Reverse_Subtract, Min, Max}
			Blend_Op selects the color or alpha blend equation.

		Blend_State :: struct {enabled: bool, src_factor: Blend_Factor, dst_factor: Blend_Factor, op: Blend_Op, src_alpha_factor: Blend_Factor, dst_alpha_factor: Blend_Factor, alpha_op: Blend_Op}
			Blend_State configures color blending for one render target.

		Buffer :: distinct u64
			GPU resource handles are opaque generational IDs. Zero is always invalid.

		Buffer_Binding :: struct {buffer: Buffer, offset: int}
			Buffer_Binding binds a buffer handle with a byte offset.

		Buffer_Desc :: struct {label: string, usage: Buffer_Usage, size: int, storage_stride: int, data: Range}
			Buffer_Desc creates a Buffer. storage_stride > 0 creates a structured storage buffer.

		Buffer_Read_Desc :: struct {buffer: Buffer, offset: int, data: Range}
			Buffer_Read_Desc describes a synchronous GPU-to-CPU buffer readback.

		Buffer_State :: struct {valid: bool, usage: Buffer_Usage, size: int, storage_stride: int}
			Buffer_State is returned by query_buffer_state for validation and diagnostics.

		Buffer_Update_Desc :: struct {buffer: Buffer, offset: int, data: Range}
			Buffer_Update_Desc describes a CPU-to-GPU buffer update.

		Buffer_Usage :: bit_set[Buffer_Usage_Flag]
		Buffer_Usage_Flag :: enum {Vertex, Index, Uniform, Storage, Immutable, Dynamic_Update, Stream_Update}
			Buffer_Usage_Flag describes the intended roles and CPU update path for a buffer.

		Color :: struct {r: f32, g: f32, b: f32, a: f32}
			Color is a linear RGBA color value.

		Color_Attachment_Action :: struct {load_action: Load_Action, store_action: Store_Action, clear_value: Color}
			Color_Attachment_Action controls load/store behavior for one color attachment.

		Color_Attachment_View_Desc :: struct {image: Image, format: Pixel_Format, mip_level: i32, layer: i32}
			Color_Attachment_View_Desc creates a render-target attachment view.

		Color_State :: struct {write_mask: u8, blend: Blend_State}
			Color_State configures write mask and blending for one color attachment slot.

		Compare_Func :: enum {Always, Never, Less, Less_Equal, Equal, Greater_Equal, Greater, Not_Equal}
			Compare_Func selects depth comparison behavior.

		Compute_Pass_Desc :: struct {label: string}
			Compute_Pass_Desc begins a compute-only pass.

		Compute_Pipeline :: distinct u64
		Compute_Pipeline_Desc :: struct {label: string, shader: Shader, pipeline_layout: Pipeline_Layout}
			Compute_Pipeline_Desc creates an immutable compute pipeline.

		Context :: struct {...}
			Context owns all GPU resources and current frame state for one backend instance.

		Cull_Mode :: enum {None, Front, Back}
			Cull_Mode selects which triangle faces are discarded.

		Depth_Attachment_Action :: struct {load_action: Load_Action, store_action: Store_Action, clear_value: f32}
			Depth_Attachment_Action controls load/store behavior for a depth attachment.

		Depth_State :: struct {format: Pixel_Format, enabled: bool, write_enabled: bool, compare: Compare_Func}
			Depth_State configures depth testing for a graphics pipeline.

		Depth_Stencil_Attachment_View_Desc :: struct {image: Image, format: Pixel_Format, mip_level: i32, layer: i32}
			Depth_Stencil_Attachment_View_Desc creates a depth-stencil attachment view.

		Desc :: struct {backend: Backend, width: i32, height: i32, native_window: rawptr, swapchain_format: Pixel_Format, vsync: bool, debug: bool, label: string}
			Desc configures a graphics Context.

		Error_Code :: enum {None, Validation, Unsupported, Invalid_Handle, Wrong_Context, Stale_Handle, Backend, Device_Lost, Resource_Leak}
			Error_Code classifies the most recent context error for programmatic handling.

		Error_Info :: struct {code: Error_Code, message: string}
			Error_Info groups the most recent context error code and human-readable message.

		Face_Winding :: enum {Clockwise, Counter_Clockwise}
			Face_Winding selects the front-face winding order.

		Features :: struct {backend: Backend, render_to_texture: bool, multiple_render_targets: bool, msaa_render_targets: bool, depth_attachment: bool, depth_only_pass: bool, sampled_depth: bool, storage_images: bool, storage_buffers: bool, compute: bool, dynamic_textures: bool, mipmapped_textures: bool, buffer_updates: bool, buffer_readback: bool}
			Features reports optional behavior supported by the active backend.

		Fill_Mode :: enum {Solid, Wireframe}
			Fill_Mode selects solid or wireframe rasterization.

		Filter :: enum {Nearest, Linear}
			Filter selects nearest or linear sampling for a Sampler.

		Image :: distinct u64
		Image_Desc :: struct {label: string, kind: Image_Kind, usage: Image_Usage, width: i32, height: i32, depth: i32, mip_count: i32, array_count: i32, sample_count: i32, format: Pixel_Format, data: Range, mips: [MAX_IMAGE_MIPS]Image_Subresource_Data}
			Image_Desc creates an Image resource and optional immutable initial contents.

		Image_Kind :: enum {Image_2D, Image_3D, Cube}
			Image_Kind describes the dimensional shape of an image resource.

		Image_Resolve_Desc :: struct {source: Image, destination: Image}
			Image_Resolve_Desc resolves a multisampled color image into a single-sampled image.

		Image_State :: struct {valid: bool, kind: Image_Kind, usage: Image_Usage, width: i32, height: i32, depth: i32, mip_count: i32, array_count: i32, sample_count: i32, format: Pixel_Format}
			Image_State is returned by query_image_state for validation and diagnostics.

		Image_Subresource_Data :: struct {data: Range, row_pitch: i32, slice_pitch: i32}
			Image_Subresource_Data supplies initial data for one image mip level.

		Image_Update_Desc :: struct {image: Image, mip_level: i32, array_layer: i32, x: i32, y: i32, width: i32, height: i32, data: Range, row_pitch: i32}
			Image_Update_Desc describes a CPU-to-GPU image subregion update.

		Image_Usage :: bit_set[Image_Usage_Flag]
		Image_Usage_Flag :: enum {Texture, Storage_Image, Color_Attachment, Depth_Stencil_Attachment, Immutable, Dynamic_Update, Stream_Update}
			Image_Usage_Flag describes sampled, storage, attachment, and CPU update roles for an image.

		Index_Type :: enum {None, Uint16, Uint32}
			Index_Type selects the element format for the bound index buffer.

		Layout_Desc :: struct {buffers: [MAX_VERTEX_BUFFERS]Vertex_Buffer_Layout, attrs: [MAX_VERTEX_ATTRIBUTES]Vertex_Attribute_Desc}
			Layout_Desc groups vertex buffer layouts and attributes for a graphics pipeline.

		Limits :: struct {max_vertex_buffers: int, max_vertex_attributes: int, max_color_attachments: int, max_resource_views: int, max_samplers: int, max_uniform_blocks: int, max_binding_groups: int, max_shader_bindings: int, max_image_mips: int, max_image_dimension_2d: int, max_image_array_layers: int, max_image_sample_count: int, max_compute_thread_groups_per_dimension: int}
			Limits reports public API limits plus backend-specific limits when queried from a Context.

		Load_Action :: enum {Clear, Load, Dont_Care}
			Load_Action selects how an attachment is initialized when a pass begins.

		Pass_Action :: struct {colors: [MAX_COLOR_ATTACHMENTS]Color_Attachment_Action, depth: Depth_Attachment_Action, stencil: Stencil_Attachment_Action}
			Pass_Action groups all attachment load/store actions for begin_pass.

		Pass_Desc :: struct {label: string, color_attachments: [MAX_COLOR_ATTACHMENTS]View, depth_stencil_attachment: View, action: Pass_Action}
			Pass_Desc begins a render pass. With no explicit attachments, the pass targets the context's implicit window swapchain.

		Pipeline :: distinct u64
		Pipeline_Desc :: struct {label: string, shader: Shader, pipeline_layout: Pipeline_Layout, primitive_type: Primitive_Type, index_type: Index_Type, layout: Layout_Desc, color_formats: [MAX_COLOR_ATTACHMENTS]Pixel_Format, depth_only: bool, colors: [MAX_COLOR_ATTACHMENTS]Color_State, depth: Depth_State, raster: Raster_State}
			Pipeline_Desc creates an immutable graphics pipeline.

		Pipeline_Layout :: distinct u64
		Pipeline_Layout_Desc :: struct {label: string, group_layouts: [MAX_BINDING_GROUPS]Binding_Group_Layout}
			Pipeline_Layout_Desc composes generated binding group layouts for one pipeline family.

		Pixel_Format :: enum {Invalid, RGBA8, BGRA8, RGBA16F, RGBA32F, R32F, D24S8, D32F}
			Pixel_Format describes image, view, attachment, and storage image formats.

		Primitive_Type :: enum {Triangles, Lines, Points}
			Primitive_Type selects the topology used by a graphics pipeline.

		Range :: struct {ptr: rawptr, size: int}
			Range is a raw byte span used for uploads, readback, and shader bytecode.

		Raster_State :: struct {fill_mode: Fill_Mode, cull_mode: Cull_Mode, winding: Face_Winding}
			Raster_State configures rasterization for a graphics pipeline.

		Sampler :: distinct u64
		Sampler_Desc :: struct {label: string, min_filter: Filter, mag_filter: Filter, mip_filter: Filter, wrap_u: Wrap, wrap_v: Wrap, wrap_w: Wrap}
			Sampler_Desc creates a texture sampler state.

		Shader :: distinct u64
		Shader_Binding_Desc :: struct {active: bool, stage: Shader_Stage, kind: Shader_Binding_Kind, group: u32, slot: u32, native_slot: u32, native_space: u32, name: string, size: u32, view_kind: View_Kind, access: Shader_Resource_Access, storage_image_format: Pixel_Format, storage_buffer_stride: u32}
			Shader_Binding_Desc carries Slang-reflected binding metadata for runtime validation.

		Shader_Binding_Kind :: enum {Uniform_Block, Resource_View, Sampler}
			Shader_Binding_Kind identifies reflected shader binding categories.

		Shader_Desc :: struct {label: string, stages: [3]Shader_Stage_Desc, has_binding_metadata: bool, bindings: [MAX_SHADER_BINDINGS]Shader_Binding_Desc, has_vertex_input_metadata: bool, vertex_inputs: [MAX_VERTEX_ATTRIBUTES]Shader_Vertex_Input_Desc}
			Shader_Desc creates a backend shader object from compiled stage bytecode and metadata.

		Shader_Resource_Access :: enum {Unknown, Read, Write, Read_Write}
			Shader_Resource_Access describes reflected read/write intent for resource views.

		Shader_Stage :: enum {Vertex, Fragment, Compute}
			Shader_Stage identifies one compiled shader stage inside a Shader_Desc.

		Shader_Stage_Desc :: struct {stage: Shader_Stage, entry: string, bytecode: Range}
			Shader_Stage_Desc provides backend bytecode for one shader stage.

		Shader_Stage_Set :: bit_set[Shader_Stage]
			Shader_Stage_Set is a bit set of shader stages used by reflected layouts.

		Shader_Vertex_Input_Desc :: struct {active: bool, semantic: string, semantic_index: u32, format: Vertex_Format}
			Shader_Vertex_Input_Desc carries Slang-reflected vertex input metadata.

		Stencil_Attachment_Action :: struct {load_action: Load_Action, store_action: Store_Action, clear_value: u8}
			Stencil_Attachment_Action controls load/store behavior for a stencil attachment.

		Storage_Buffer_View_Desc :: struct {buffer: Buffer, offset: int, size: int}
			Storage_Buffer_View_Desc creates a storage buffer byte range view.

		Storage_Image_View_Desc :: struct {image: Image, format: Pixel_Format, mip_level: i32, base_layer: i32, layer_count: i32}
			Storage_Image_View_Desc creates a storage image view.

		Store_Action :: enum {Store, Dont_Care}
			Store_Action selects whether an attachment value is preserved when a pass ends.

		Texture_View_Desc :: struct {image: Image, format: Pixel_Format, base_mip: i32, mip_count: i32, base_layer: i32, layer_count: i32}
